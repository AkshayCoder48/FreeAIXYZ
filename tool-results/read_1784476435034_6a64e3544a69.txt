   773→   773→/** Find a model by id (case-insensitive). Returns undefined if not found. */
   774→   774→export function findModel(id: string | undefined): GatewayModel | undefined {
   775→   775→  if (!id) return undefined;
   776→   776→  const lower = id.toLowerCase();
   777→   777→  return MODELS.find((m) => m.id.toLowerCase() === lower);
   778→   778→}
