   805→/** Find a model by id (case-insensitive). Returns undefined if not found. */
   806→export function findModel(id: string | undefined): GatewayModel | undefined {
   807→  if (!id) return undefined;
   808→  const lower = id.toLowerCase();
   809→  return MODELS.find((m) => m.id.toLowerCase() === lower);
   810→}
